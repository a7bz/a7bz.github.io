vue2 自动路由 ​

每次在 views 下添加页面后，都需要去配置路由，总是手动去设置，会比较麻烦

于是研究了一下根据目录结构自动生成路由，这样在 views 下添加文件之后就会自动生成路由了

const routes = [

      },
      {
        path: "/test",
        name: "test",
        component: () => import("@/views/test/index.vue"),
      },
      {
        path: "/test/index",
        name: "test-test",
        component: () => import("@/views/test/test.vue"),

];

首先删除尾部的.vue 通过.slice(0, -4)

const pages = glob.keys().map((item) => {
  const name = item.slice(0, -4);
  const x = name.split("/");
  const page = {};
  if (x.length == 2) {
    if (x[1] == "index") {
      page.path = ``;
      page.name = "index";
    } else {
      page.path = `/${x[1]}`;
      page.name = x[1];
    }
  } else {
    if (x[2] == "index") {
      page.path = `/${x[1]}`;

      page.name = x[1] + "-" + x[2];
    }
  }
  page.component = glob(item).default;
  return page;
});

router/index.js

js
import Vue from "vue";
import VueRouter from "vue-router";

Vue.use(VueRouter);

const glob = require.context("@/views", true, /\.vue$/);
const pages = glob.keys().map((item) => {
  const name = item.slice(0, -4);
  const x = name.split("/");
  const page = {};
  if (x.length == 2) {
    if (x[1] == "index") {
      page.path = ``;
      page.name = "index";
    } else {
      page.path = `/${x[1]}`;
      page.name = x[1];
    }
  } else {
    if (x[2] == "index") {
      page.path = `/${x[1]}`;
      page.name = x[1];
    } else {
      page.path = `/${x[1]}/${x[2]}`;
      page.name = x[1] + "-" + x[2];
    }
  }
  page.component = glob(item).default;
  return page;
});
const routes = [
  {
    path: "/",
    component: () => import("@/layout/index.vue"),
    children: pages,
  },
];

const router = new VueRouter({
  mode: "history",
  base: process.env.BASE_URL,
  routes,
});

export default router;
